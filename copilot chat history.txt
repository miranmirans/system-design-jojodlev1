function getCharacterOfTheDay(characters) {
    // Get current date in UTC (YYYY-MM-DD format) to ensure global consistency
    const today = new Date();
    const dateString = today.getUTCFullYear() + '-' + 
                      String(today.getUTCMonth() + 1).padStart(2, '0') + '-' + 
                      String(today.getUTCDate()).padStart(2, '0');
    
    // Create a simple hash from the date string
    // This creates a deterministic "seed" from the date
    let hash = 0;
    for (let i = 0; i < dateString.length; i++) {
        const char = dateString.charCodeAt(i);
        hash = ((hash << 5) - hash) + char;
        hash = hash & hash; // Convert to 32-bit integer
    }
    
    // Use the hash to select an index
    // Make sure it's positive and within array bounds
    const index = Math.abs(hash) % characters.length;
    
    return characters[index];
}

User: What is a modulo operator?

GitHub Copilot: The modulo operator (`%`) returns the **remainder** after division.

- `10 % 3 = 1` (10 divided by 3 is 3 with remainder 1)
- `15 % 4 = 3` (15 divided by 4 is 3 with remainder 3)
- `8 % 2 = 0` (8 divided by 2 is 4 with no remainder)

const index = Math.abs(hash) % characters.length;

The modulo operator effectively converts any number into a valid array index (0 to length-1), which is why it's perfect for selecting a random-seeming but deterministic element from an array.
